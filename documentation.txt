ChatGpt example of WordPress DockerFile:
# Use an official WordPress image as the base image
FROM wordpress:latest

# Install additional dependencies
RUN apt-get update && apt-get install -y \
    php-cgi \
    php7.3 \
    php-fpm \
    php-mysql \
    php-pdo \
    php-gd \
    php-cli \
    php-mbstring \
    wget \
    tar \
    curl \
    bash \
    mariadb-client

# Install WP-CLI
RUN curl -O https://raw.githubusercontent.com/wp-cli/builds/gh-pages/phar/wp-cli.phar && \
    chmod +x wp-cli.phar && \
    mv wp-cli.phar /usr/local/bin/wp

# Set the working directory to the WordPress installation
WORKDIR /var/www/html

# Expose the port for the web server (if needed)
# EXPOSE 80

# Start the container's main process (in this case, the WordPress site)
CMD ["php-fpm", "-F"]
